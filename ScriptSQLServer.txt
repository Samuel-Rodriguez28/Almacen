CREATE TABLE STOCK (
  idSTOCK INTEGER   NOT NULL UNIQUE,
  UNIDADES INTEGER    ,
  CODIGO INTEGER    ,
  NOM_ART VARCHAR(30)    ,
  MARCA VARCHAR(20)    ,
  OBSERVACION VARCHAR(20)    ,
  MINIMO INTEGER      ,
  MEDIDA VARCHAR(20)
PRIMARY KEY(idSTOCK));




CREATE TABLE REQUISICIONES (
  idREQUISICIONES INTEGER   NOT NULL UNIQUE,
  JEF CHAR    ,
  PLAN CHAR    ,
  SUB CHAR    ,
  DIR CHAR    ,
  COM CHAR    ,
  REC CHAR    ,
  OBSERV VARCHAR(40)    ,
  FECHA DATE      ,
PRIMARY KEY(idREQUISICIONES));




CREATE TABLE CIERRE_MATERIAL (
  idCIERRE_MATERIAL INTEGER   NOT NULL UNIQUE,
  OC INTEGER    ,
  NUM_REQ INTEGER    ,
  NUM_FAC INTEGER    ,
  PROVEEDOR VARCHAR(30)    ,
  FEC_SOL DATE    ,
  FEC_ENT DATE    ,
  NOMBRE_ARTIC VARCHAR(30)    ,
  OBSERV VARCHAR(30)      ,
PRIMARY KEY(idCIERRE_MATERIAL));




CREATE TABLE COTIZACION (
  idCOTIZACION INTEGER   NOT NULL UNIQUE,
  ARTIC VARCHAR(30)    ,
  UM VARCHAR(10)    ,
  CANT INTEGER    ,
  CLAVE INTEGER    ,
  FERR CHAR    ,
  ELECT CHAR    ,
  PLOM CHAR      ,
PRIMARY KEY(idCOTIZACION));




CREATE TABLE SALIDA (
  idSALIDA INTEGER   NOT NULL ,
  idSTOCK INTEGER   NOT NULL ,
  PZAS INTEGER    ,
  CODIGO INTEGER    ,
  NOMBRE VARCHAR(30)    ,
  PERSONAL VARCHAR(30)    ,
  DESTINO VARCHAR(20)    ,
  OBSERV VARCHAR(20)    ,
  FECHA DATE      ,
PRIMARY KEY(idSALIDA));




CREATE TABLE ENTRADA (
  idENTRADA INTEGER   NOT NULL ,
  idSTOCK INTEGER   NOT NULL ,
  UNIDADES INTEGER    ,
  NOM_ART VARCHAR(40)    ,
  ESTADO VARCHAR(20)    ,
  FECHA DATE      ,
PRIMARY KEY(idENTRADA));
